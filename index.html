Contenu de server.js (inclus dans le zip)
const express = require('express'); const stripe = require('stripe')(process.env.STRIPE_SK); const app = express(); app.use(express.json()); app.use(express.static('public')); app.post('/api/checkout', async (req, res) => { const { items, origin } = req.body; const session = await stripe.checkout.sessions.create({ payment_method_types: ['card'], line_items: items, mode: 'payment', success_url: origin + '/success.html', cancel_url: origin, }); res.json({ url: session.url }); }); app.listen(process.env.PORT || 3000, () => console.log('Live ✓'));
